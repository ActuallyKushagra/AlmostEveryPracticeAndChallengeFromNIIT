# d) Expected Value/ Mean & standard deviation

mean = 0
var = 0

for i, j in zip(X, p_x):
    mean += i*j
        
for i, j in zip(X, p_x):
    var += ((i-mean)**2)*j
    
std = math.sqrt(var)

print(f'Expected Value/ Mean is {mean} and Standard Deviation is {std}')

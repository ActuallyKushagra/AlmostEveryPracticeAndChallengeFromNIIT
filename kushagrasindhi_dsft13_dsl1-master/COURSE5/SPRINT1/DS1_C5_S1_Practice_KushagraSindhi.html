for i in df: 
    uout=[]
    lout=[]
    mi=min(df[i])
    ma=max(df[i])
    ra=ma-mi
    q1=df[i].quantile(0.25)
    q3=df[i].quantile(0.75)
    iqr=q3-q1
    lf=q1-(1.5*iqr)
    uf=q3+(1.5*iqr)
    
    for i in df[i]:
        if i > uf:
            uout.append(i)
        elif i<lf:
            lout.append(i)
            
    if software=='Statistics':
